## 1.内核&硬件

```bash
  # 远程连接服务器,27822端口，bttech用户名，server.bontor.cn服务器地址
  ssh -p 27822 bttech@server.bontor.cn
  # 查看操作系统内核
  lsb_release -a
  # 查看内核版本
  uname -a
  # 检查文件系统磁盘占用情况(以1k作为计量空间)
  df
  # 检查文件系统磁盘占用情况(以1024M|1G作为计量空间)
  df -TH
  # 去到根目录
  cd /
  # 查看目录
  ls -la
  # 查看Linux系统运行的一些进程（ctrl+c退出）
  top
```

## 2.文件

```bash
# 切换到home目录
cd /home/
# 查看目录
ls
# 创建test文件夹
mkdir test
# 创建文件
touch test.txt
# 进入vi编辑器内部（点击i，进入insert状态，使用esc退出编辑，使用:wq保存并退出，:q!不对文件进行保存）
vi test.txt
# 查看文件
cat test.txt
# 往文件添加内容
echo '123' >> test.txt
# 使用内容去覆盖文件内容
echo '123' > test.txt
# 删除文件
rm test.txt
# 删除目录
rm -r test/
# 下载资源
wget 资源地址
# 解压tar.gz
tar zxvf 文件名
# 压缩文件
tar zcvf 压缩后文件名 要压缩的文件名
# 查看docker进程（grep后面接要筛选的进程）
ps -ef | grep docker
# 强制杀死进程(-9表示强制终止)
kill -9 xxid
```

## 3.查看系统服务状态

```bash
# 查看sshd运行状态(sshd服务名字)
service sshd status
# 关闭sshd服务
service sshd stop
# 重启sshd服务
service sshd restart
# 运行防火墙(systemctl也是管理服务的命令)
systemctl status firewalld.service
# 关闭内部防火墙
systemctl stop firewalld      # 临时关
systemctl disable firewalld   # 永久关. 重启以后也是关着的

# 开启内部防火墙
#systemctl start firewalld
```

## 4.ssh秘钥的方式远程连接Linux

```bash

service shhd status
# 查看默认监听端口
netstat -anlp | grep sshd
# 查看默认监听端口
vi /etc/ssh/sshd_config
#

service sshd restart
```
